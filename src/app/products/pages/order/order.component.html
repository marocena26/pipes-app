sortBy

Sorts a set of elements in alphabetical order

{{ [object] | sortBy : orderBy }} 


          transform(heroes: Hero[], sortBy?: keyof Hero | ''): Hero[] {
            switch (sortBy) {
              case 'name':
                return heroes.sort((a, b) => (a.name > b.name ? 1 : -1));

              case 'canFly':
                return heroes.sort((a, b) => (a.canFly > b.canFly ? 1 : -1));

              case 'color':
                return heroes.sort((a, b) => (a.color > b.color ? 1 : -1));

              default:
                return heroes;
            }
          }

colorCase

Sorts a set of elements by color

{{ [object] | colorCase }} 

          transform(value: Color): string {
            switch (value) {
              case Color.red:
                return '0 - Red';
              case Color.black:
                return '1 - Black';
              case Color.blue:
                return '2 - Blue';
              case Color.green:
                return '3 - Green';
              default:
                return value;
            }
          }
